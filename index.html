import os import pandas as pd import matplotlib.pyplot as plt # 1. 파일 확인 및 데이터 불러오기 csv_file = '전력수요.csv' if not os.path.exists(csv_file): print(f"Error: {csv_file} 파일이 없습니다.") exit(1) df = pd.read_csv(csv_file) # 2. '일시' 컬럼을 날짜형으로 변환 후 인덱스로 설정 (중요!) df['일시'] = pd.to_datetime(df['일시']) df.set_index('일시', inplace=True) # 3. 깃허브 액션 그래프 생성을 위한 데이터 처리 (일별 평균) daily_df = df.resample('D').mean() # 4. 그래프 이미지 생성 및 저장 fig, ax = plt.subplots(figsize=(10, 5)) ax.plot(daily_df.index, daily_df['전력수요(MWh)'], color='#0066cc', linewidth=1.5, label='Daily Avg Demand') ax.set_title('2025 Electricity Demand Trends') ax.set_xlabel('Date') ax.set_ylabel('Power (MWh)') ax.legend() ax.grid(True, linestyle='--', alpha=0.5) plt.tight_layout() image_path = 'power_chart.png' plt.savefig(image_path, dpi=200) plt.close() # 5. """ with open('report.html', 'w', encoding='utf-8') as f: f.write(html_content) print("시각화 이미지 및 report.html 생성 완료!")
